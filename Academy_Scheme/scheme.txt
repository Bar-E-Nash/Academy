CREATE TABLE Courses(
CourseID INT PRIMARY KEY IDENTITY,
CourseName VARCHAR(20) UNIQUE,
DateOfOpen DATE,
CourseFee INT,
DaysInWeek VARCHAR(100),
Hours VARCHAR(15)
)

CREATE TABLE Students(
StudentID INT PRIMARY KEY IDENTITY,
NatID INT NOT NULL,
FirstName VARCHAR(15) NOT NULL,
LastName VARCHAR(15) NOT NULL,
Phone VARCHAR(15),
ToPay INT
)

CREATE TABLE Teachers(
TeacherID INT PRIMARY KEY IDENTITY,
HourlyRate INT,
FirstName VARCHAR NOT NULL,
LastName VARCHAR NOT NULL
)

CREATE TABLE Topics
(
TopicID INT PRIMARY KEY IDENTITY,
TopicName VARCHAR(20) NOT NULL,
ParentCourse INT FOREIGN KEY REFERENCES Courses(CourseID),
Teacher INT FOREIGN KEY REFERENCES Teachers(TeacherID),
TopicHours int
)

CREATE TABLE Tuitions(
StudentID INT FOREIGN KEY REFERENCES Students(StudentID),
Tuition INT
)
